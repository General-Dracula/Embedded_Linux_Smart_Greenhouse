70int I2CDevice::open(){
71 string name;
72
73 switch(this->bus)
74 {
75 case 0:
76 name = BBB_I2C_0;
77 break;
78 case 1:
79 name = BBB_I2C_1;
80 break;
81 default:
82 name = BBB_I2C_2;
83 break;
84 }
85
86 if((this->file=::open(name.c_str(), O_RDWR)) < 0){
87 perror("I2C: failed to open the bus\n");
88 return 1;
89 }
90 if(ioctl(this->file, I2C_SLAVE, this->device) < 0){
91 perror("I2C: Failed to connect to the device\n");
92 return 1;
93 }
94 return 0;
95}

136unsigned char I2CDevice::readRegister(unsigned int registerAddress){

154unsigned char* I2CDevice::readRegisters(unsigned int number, unsigned int fromAddress){
155 this->write(fromAddress);
156 unsigned char* data = new unsigned char[number];
157 if(::read(this->file, data, number)!=(int)number){
158 perror("IC2: Failed to read in the full buffer.\n");
159 return NULL;
160 }
161 return data;
162}
